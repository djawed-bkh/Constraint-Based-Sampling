    public void jaccardCDFItemsetRev(int k, Database db, String outputFilePath) {
        ArrayList<IP> patterns = new ArrayList<>();
        double[] thresholds = {0.2, 0.4, 0.6, 0.8, 1};
        HashMap<Double, Double> results = new HashMap<>();

        // 1- Draw k patterns
        ItemsetSampling its = new ItemsetSampling(db);
        for (int i = 0; i < k; i++) {
            patterns.add(its.drawItemsetToIP());
        }

        // 2- Calculate the CDF
        for (double thresh : thresholds) {
            int count = 0;
            for (int i = 0; i < patterns.size(); i++) { // iterate over each unique pair of patterns
                for (int j = i + 1; j < patterns.size(); j++) {
                    double jacc = jaccard(patterns.get(i), patterns.get(j));
                    if (jacc <= thresh) {
                        count++;
                    }
                }
            }
            double cdfValue = (double) count * 2 / (k * (k - 1));
            results.put(thresh, cdfValue);
        }

        // 3- Write the results to a CSV file
        try (BufferedWriter writer = new BufferedWriter(new FileWriter(outputFilePath))) {
            // Write the header
            writer.write("Jaccard threshold,CDF Value");
            writer.newLine();

            // Write each threshold and its corresponding CDF value
            for (double thresh : thresholds) {
                writer.write(thresh + "," + results.get(thresh));
                writer.newLine();
            }
        } catch (IOException e) {
            e.printStackTrace();
        }
    }

                int indexub = (value + (int) (Math.random() * ((dataset.getDistinctValues().get(m).size()) - value))); //WARNING: works only for normalized data

public void CFIPSNumberOfDrawsEvaluation(Database db, int numberOfPatterns, int NumberOfConstraints, int generatorType, int rep) {
        ArrayList<Constraint> constraintList = new ArrayList<>();
        CFIPS CFIPS = new CFIPS(db, constraintList);

        if (generatorType == 1) {     // Totaly random constraints
            ConstraintGenerator constgen = new ConstraintGenerator(db, NumberOfConstraints, 1);
            constraintList = constgen.TotalyRandomConstraints(NumberOfConstraints);
        } else if (generatorType == 2) { // Ascending order of attribute constraints
            ConstraintGenerator constgen = new ConstraintGenerator(db, NumberOfConstraints, 2, true);
            constraintList = constgen.RandomConstraintsOrderedAttributes(NumberOfConstraints, true);
        } else if (generatorType == 3) {
            ConstraintGenerator constgen = new ConstraintGenerator(db, NumberOfConstraints, 2, false);
            constraintList = constgen.TotalyRandomConstraints(NumberOfConstraints);
        } else {      //4 user based

        }


        double meanDraws = 0.0;
        long expestart = System.currentTimeMillis();
        long timeout = 300000; // 5 minutes in milliseconds
        long TotalCpuTime = 0;
        for (int r = 0; r < rep; r++) {
            int i = 0;
            int Ndraws = 0;
            if (System.currentTimeMillis() - expestart > timeout) {
                System.out.println("Timeout occurred. Exiting the process.");
                return; // Exit when timeout
            }


            // Track the start time
            long startTime = System.currentTimeMillis();


            while (i < numberOfPatterns) {
                // Check if the elapsed time has exceeded the timeout
                if (System.currentTimeMillis() - startTime > timeout) {
                    System.out.println("Timeout occurred. Exiting the process.");
                    System.out.println("Number of Iterations before timeout: " + Ndraws);
                    return; // Exit the method if the timeout is reached
                }

                long drawStart = System.currentTimeMillis();

                IP ip = CFIPS.drawIP();

                TotalCpuTime += System.currentTimeMillis() - drawStart;
                Ndraws++;

                boolean satisfyConst = true;
                for (Constraint c : constraintList) { // ici check le respect des contraintes
                    if (c instanceof Inclusion) {
                        if (!checkInclusionConstraint(c.getMprim(), c.getVprim(), ip)) {
                            satisfyConst = false;
                            break;
                        }

                    } else if (c instanceof Exclusion) {
                        if (!checkExclusionConstraint(c.getMprim(), c.getVprim(), ip)) {
                            satisfyConst = false;
                            break;
                        }
                    } else if (c instanceof Sup) {
                        if (!checkSupConstraint(c.getMprim(), c.getVprim(), ip)) {
                            satisfyConst = false;
                            break;
                        }
                    } else if (c instanceof Inf) {
                        if (!checkInfConstraint(c.getMprim(), c.getVprim(), ip)) {
                            satisfyConst = false;
                            break;
                        }
                    } else if (c instanceof SupEq) {
                        if (!checkSupEqConstraint(c.getMprim(), c.getVprim(), ip)) {
                            satisfyConst = false;
                            break;
                        }
                    } else if (c instanceof InfEq) {
                        if (!checkInfEqConstraint(c.getMprim(), c.getVprim(), ip)) {
                            satisfyConst = false;
                            break;
                        }
                    }
                }
                if (satisfyConst) {
                    i++;
                }
            }
            meanDraws += Ndraws;
        }
        double result = meanDraws / rep;
        System.out.println("Constrained FIPS: " + numberOfPatterns + " patterns are drawn in " + result + " iterations");
        System.out.println("Gain(%): " + result / numberOfPatterns);


    }

        public IP ProcessClosure(IP ip){

               int lowerBound = Integer.MAX_VALUE;
               int upperBound = Integer.MIN_VALUE;
               for(int g : ip.getCoverage()){
                if(dataset.getObject(g).get(m) < lowerBound ){
                    lowerBound= dataset.getObject(g).get(m);
                }
                   if(dataset.getObject(g).get(m) > upperBound ){
                       upperBound= dataset.getObject(g).get(m);
                   }
               }
               ClosedIP.add(lowerBound);
               ClosedIP.add(upperBound);
           }

            return new IP(ClosedIP, dataset);
        }

           for(IP pattern: patternPool){

                       avgDifference += diff;                // ajouter la difference de support dans avgDifference
                       avgPrecision += frequency;
                   }
               }
           }

           avgPrecision = avgDifference / avgPrecision;

public class FIPS {

    protected final Database dataset;
    ArrayList<Double> CumulativeObjectsweights = new ArrayList<>();
    double ObjectstotalWeight = 0.0;

    public FIPS(Database dataset) {
        this.dataset = dataset;

        ProcessObjectsWeights();

    }



    public void ProcessObjectsWeights(){

        for(int g = 0; g< this.dataset.getObjectNumber(); g++){
            double objectWeight=1.0;
            for(int m = 0; m< this.dataset.getColumnsNumberNumerical(); m++){

                int valIndex= this.dataset.getObject(g).get(m);
                if(valIndex == -1){
                    System.out.println("ERROR: We should not be here");
                }

                objectWeight *= valIndex + (dataset.getDistinctValues().get(m).size() - valIndex) + ( valIndex * (dataset.getDistinctValues().get(m).size() - valIndex ))  + 1 ;
            }
            if(CumulativeObjectsweights.isEmpty()){
                CumulativeObjectsweights.add(objectWeight);

            }else{
                CumulativeObjectsweights.add(CumulativeObjectsweights.get(CumulativeObjectsweights.size()-1) + objectWeight);
            }

            ObjectstotalWeight += objectWeight;
        }
    }

    public void ProcessTransactionsWeights(){

        for(int g = 0; g< this.dataset.getObjectNumber(); g++){
            int Nitems=0;
            for(Integer bin: this.dataset.getTrasaction(g) ){
                Nitems += bin;
            }
            double objectWeight = Math.pow(2, Nitems);

            if(Cumulativetransactionsweights.isEmpty()){
                Cumulativetransactionsweights.add(objectWeight);

            }else{
                Cumulativetransactionsweights.add(Cumulativetransactionsweights.get(Cumulativetransactionsweights.size()-1) + objectWeight);
            }
            TransactionstotalWeight += objectWeight;
        }
    }



    public int DrawTransaction(){
        double drawnobject= Math.random() * TransactionstotalWeight;
        return FindIndex(0, dataset.getObjectNumber(), drawnobject);
    }

    private int FindIndex(int left, int right, double drawnobject) {
        int middle = (left + right) / 2;
        if (Cumulativetransactionsweights.get(middle) >= drawnobject)
            if (middle == 0 || Cumulativetransactionsweights.get(middle-1) < drawnobject)
                return middle;
            else
                return FindIndex(left, middle, drawnobject);
        else
            return FindIndex(middle, right, drawnobject);
    }

    public static List<List<Integer>> transpose(List<List<Integer>> database) {
        List<List<Integer>> TransposDatabase = new ArrayList<>();

        if (database.isEmpty()) {
            return TransposDatabase; // Return empty list if input is empty
        }

        int rowCount = database.size();
        int colCount = database.get(0).size();

        // Initialize the transposed list with empty lists
        for (int i = 0; i < colCount; i++) {
            TransposDatabase.add(new ArrayList<>());
        }

        // Fill the transposed list
        for (int i = 0; i < rowCount; i++) {
            for (int j = 0; j < colCount; j++) {
                TransposDatabase.get(j).add(database.get(i).get(j));
            }
        }

        return TransposDatabase;
    }
